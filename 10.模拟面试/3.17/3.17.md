         	var temp = function() {};

             	var _this = this innstanceof fn ? this : context;
             	fn.apply(_this, [...args, ...innerArgs])
           }
         
         	temp.prototype = fn.prototype;
         
         	inner.prototype = new temp();
         
         	return inner;
